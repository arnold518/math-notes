\documentclass{beamer}
\usetheme{Madrid}  % Use the Madrid theme
\usepackage{amsmath, amssymb}

\definecolor{mytheorem}{RGB}{38, 38, 134} % Dark blue

\newenvironment{theoremblock}[1]{%
    \setbeamercolor{block title}{bg=mytheorem,fg=white}  % Blue title (like block)
    \setbeamercolor{block body}{bg=blue!10,fg=black}   % Light blue body
    \begin{block}{#1}}{\end{block}}

\newenvironment{definitionblock}[1]{%
    \setbeamercolor{block title}{bg=red!70,fg=white}  % Red title (like alertblock)
    \setbeamercolor{block body}{bg=red!10,fg=black}   % Light red body
    \begin{block}{#1}}{\end{block}}

\newenvironment{conceptblock}[1]{%
    \setbeamercolor{block title}{bg=purple!70,fg=white}  % Purple title (Custom)
    \setbeamercolor{block body}{bg=purple!10,fg=black}   % Light purple body
    \begin{block}{#1}}{\end{block}}

    \begin{theoremblock}{Theorem 2.5 (Fundamental Theorem of Calculus)}
        If \( f \) is continuous on \( [a, b] \) and \( F \) is its antiderivative, then:
        \[
        \int_a^b f(x)dx = F(b) - F(a).
        \]
    \end{theoremblock}

    \vspace{10pt}

    % Custom Definition Block (Like `alertblock`)
    \begin{definitionblock}{Definition 1.3 (Prime Number)}
        A prime number is a natural number greater than \( 1 \) that has no divisors other than \( 1 \) and itself.
    \end{definitionblock}

    \vspace{10pt}

    % Custom Concept Block (New!)
    \begin{conceptblock}{Concept 3.1 (Divisibility)}
        A number \( a \) is divisible by \( b \) if there exists an integer \( k \) such that \( a = bk \).
    \end{conceptblock}

\end{frame}
